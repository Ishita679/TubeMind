% Project Overview
\begin{frame}{Project Overview}
    \begin{itemize}
        \item AI-powered web application that extracts transcripts from YouTube videos.
        \item Automatically processes and summarizes long-form video content.
        \item Helps students and professionals quickly understand lengthy lectures and tutorials.
        \item Clean, responsive, and user-friendly interface.
    \end{itemize}
\end{frame}

% Objectives
\begin{frame}{Objectives}
    \begin{itemize}
        \item Extract transcripts from YouTube video URLs.
        \item Generate concise and structured summaries using Large Language Models (LLMs).
        \item Provide fast and accurate results.
        \item Build a scalable and efficient full-stack application.
    \end{itemize}
\end{frame}

% Technology Stack
\begin{frame}{Technology Stack}
    \textbf{Frontend:} React.js, HTML, CSS, JavaScript \\[5pt]
    \textbf{Backend:} Node.js, Express.js \\[5pt]
    \textbf{AI Integration:} OpenAI API (GPT model) \\[5pt]
    \textbf{Transcript Fetching:} YouTube Transcript API \\[5pt]
    \textbf{HTTP Requests:} Axios \\[5pt]
    \textbf{Deployment:} Vercel / Render
\end{frame}

% System Architecture
\begin{frame}{System Architecture}
    \begin{enumerate}
        \item User enters YouTube video URL.
        \item Backend extracts Video ID.
        \item Transcript is fetched using Transcript API.
        \item Transcript is sent to OpenAI API.
        \item Generated summary is returned to frontend.
        \item Summary is displayed to the user.
    \end{enumerate}
\end{frame}
